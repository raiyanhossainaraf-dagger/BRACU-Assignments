public class Employee{
  
  public String name;
  public String salary="30000.0 Tk";
  public String designation="junior";
  
  public void displayInfo(){
    
    System.out.println("Employee Name:"+name );
    System.out.println("Employee Salary:"+salary );
    System.out.println("Employee Designation:"+ designation);
    
    
  }
  public void calculateTax(){
    
    if(designation.equals("junior")){
      
      System.out.println("No need to pay tax");
    }
    else if(designation.equals("lead")){
      
      System.out.println(name+" Tax Amount: 24000.00 Tk");
    }
    else if(designation.equals("manager")){
      
      System.out.println(name+" Tax Amount: 31500.00 Tk");
    }
    
  }
  
  public void promoteEmployee(String a){
    
    if(a.equals("lead")){
      designation="lead";
      salary="80000.00 Tk";
      System.out.println(name+" has been promoted to lead");

    }
    
    else if(a.equals("manager")){
      designation="manager";
      salary="105000.00 Tk";
      System.out.println(name+" has been promoted to manager");
      System.out.println("New Salary: 105000.00 Tk");
    }
    
  }

    name = a;
    
  }
