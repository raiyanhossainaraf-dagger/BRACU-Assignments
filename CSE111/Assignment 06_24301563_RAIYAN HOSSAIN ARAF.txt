public class Book{
 
  static int total_books_sold ;
  static double total_revenue;
  String title;
 
  double pd;
 
  public Book(String a, int b){
   
    title=a;
    pd=pad(b);
    total_books_sold+=1;
    total_revenue+=pd;
   
  }
 
  public void bookDetails(){
   
    System.out.println("Title: "+title);
    System.out.println("Price after Discount: "+pd);
  }
  public double pad(int a){
   
    double p=(double)(150.0-(150.0*(a/100.0)));
    return p;
  }
}

public class Borrower{
  public static int book_count[] = {3, 3, 3};
  public static String book_name[] = {"Pather Panchali", "Durgesh Nandini", "Anandmath"};
  String name;
  String [] books =new String [3];
  int c=0;
  
  public Borrower(String a){
    
    name=a;
    
  }
  
  public static void bookStatus(){
    
    System.out.println("Available Books: ");
    System.out.println(book_name[0]+": "+book_count[0]);
    System.out.println(book_name[1]+": "+book_count[1]);
    System.out.println(book_name[2]+": "+book_count[2]);
    
  }
  
  public void borrowBook(String a){
     
    if(a.equals("Pather Panchali")){
      if(book_count[0]>0){
      books[c]=a;
      c++;
      book_count[0]-=1;
      }
      else{
      System.out.println("This book is not available.");
      }
    }
    else if(a.equals("Durgesh Nandini")){
      if(book_count[1]>0){
      books[c]=a;
      c++;
      book_count[1]-=1;
    }
       else{
      System.out.println("This book is not available.");
      }
    }
    else{
       if(book_count[2]>0){
      books[c]=a;
      c++;
      book_count[2]-=1;
    }
    else{
      System.out.println("This book is not available.");
      }
    }
    
  }
  
  public void borrowerDetails(){
    
    System.out.println("Name: "+name);
    System.out.println("Books Borrowed:");
    for(int i=0;i<c;i++){
      System.out.println(books[i]);
    }
    
  }
  
  public static int remainingBooks(String a){
    
    if(a.equals("Pather Panchali")){
      return book_count[0];
    }
    else if(a.equals("Durgesh Nandini")){
      return book_count[1];
    }
    else{
      return book_count[2];
    }
    
  }
  
}

public class Cargo{
  public static int id;
  public static double capacity=10.0;
  String content;
  int Id;
  double w;
  boolean loaded=false;
  
  public Cargo(String a,double b){
  Id=id+1;
  id++;
  content =a;
  w=b;
  
  }
  
  public void details (){ 
  
    System.out.println("Cargo ID: "+Id+" Contents: "+content+" Weight: "+w+" Loaded: "+loaded);
  
  }
  public void load(){
    if(capacity-w>=0){
    loaded=true;
    System.out.println("Cargo "+Id+" loaded for transport.");
    capacity-=w;
    }
    else{
    
      System.out.println("Cannot load cargo, exceeds weight capacity.");
    
    }
  }
  
  public static double capacity(){
  return capacity;
  }
  
  public void unload(){
  
  loaded=false;
  capacity+=w;
    
  }

}

public class Player{
 
  public static int total;
  public static String []names=new String[11];
  String name;
  String country;
  int jersey;
  public Player(String a,String b,int c){
  name=a;
  country=b;
  jersey=c;
  names[total]=a;
  total+=1;
  }
 
  public static void info(){
  System.out.println("Total number of players: "+total);
  System.out.print("Players enlisted so far: ");
  for(int i=0;i<total;i++){
    if(i==total-1){
    System.out.println(names[i]);
    }
    else{
      System.out.print(names[i]+", ");
    }
  }
  }
 
  public String player_detail(){
 
   return ("Player Name: "+name+"\n"+"Jersey Number: "+jersey+"\n"+"Country: "+country);
   
  }
 
}
