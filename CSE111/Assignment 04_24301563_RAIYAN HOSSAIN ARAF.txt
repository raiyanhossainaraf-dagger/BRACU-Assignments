    this.l=l;

   public Shape2D(int h,int l,String a){
    this.h=h;
    this.l=l;

    public Shape2D(int l,int w,int h){
    this.l=l;
    this.w=w;
    this.h=h;
    shape="Triangle";
    System.out.println("A Triangle has been created with the following sides: " +l+", "+w+", "+h);
   
  }
 
  public void area(){
   
    if(shape.equals("Square")){
     double area =l*l;
      System.out.println("The area of the Square is: "+area);
     
    }
       
       else if(shape.equals("Rectangle")){
         double area =l*w;
         System.out.println("The area of the Rectangle is: "+area);
         
       }
               else if(shape.equals("Triangle")){
                 if(w==0){

                 System.out.println("The area of the Triangle is: "+area);

                 else{
                   double s= (l+h+w)/2;
                 double area =Math.sqrt(s*(s-l)*(s-w)*(s-h));
                 System.out.println("The area of the Triangle is: "+area);
                 }
               }
                       
                       }

public class Student{
  int id;

  String[]course;
  int c=0;
  public Student(int a){
    id=a;
    System.out.println("A student with ID "+a+" has been created.");
    course = new String[3];
  }
  public Student(int a,double b){
    id=a;
    cg=b;
    System.out.println("A student with ID  "+a+" and cgpa "+b+ " has been created.");
    if(b>=3.0){
      course=new String[4]; 
    } 
    else{
      course=new String[3]; 
    }
  }
  public void addCourse(String a){
    
    if(cg==0.0){
      System.out.println("Failed to add "+a);
      System.out.println("Set CG first");
      
    }
    else if(c>=course.length){
      
      System.out.println("Failed to add "+a);
      System.out.println("CG is low. Can't add more than 3 courses.");
    }
    
    else{
      course[c]=a;
      c++;
      
    }
  }
  
  public void storeCG(double cg){
    
    this.cg=cg;
    if(cg>=3){
      
      course=new String[4];
      
    }
    else{
      course=new String[3];
      
    }
    c=0;
  }
  public void addCourse(String[] a){
    
    if(cg==0.0){
      System.out.println("Failed to add "+a+" Set CG first");
      
    }
    else if(c+a.length>course.length){
      int canAdd=course.length-c; 
      for (int i=0;i<canAdd;i++){
        course[c]=a[i];
        c++;
      }
      System.out.println("Failed to add " +a[canAdd]); 
      System.out.println("Maximum "+ course.length+" courses allowed.");
    } 
    else {
      for(int i=0;i<a.length;i++){
        course[c]=a[i];
        c++;
      }
    }
  }
  public void showAdvisee(){
    
    System.out.println("Student ID: "+id+", CGPA: "+cg);
    
    if(c==0){
      System.out.println("No courses added.");
    }

        System.out.print(course[i]+" " );
      }System.out.println("");
    }

      course[i]=null;
    }

  }

public class Triangle{
  int a,b,c;
  int pre;
  
  public void updateSides(int a, int b,int c){
    
    this.a=a;
    this.b=b;
    this.c=c;
    this.pre=(a+b+c);
  }
  
  public void triangleDetails(){
    
    System.out.println("Three sides of the triangle are: "+a+","+b+","+c);
    System.out.println("Perimeter: "+pre); 
  }
  
  public String printTriangleType(){
    String g;
    if(a==b&&b==c){
      g="This is an Equilateral Triangle.";
    }
    else if(a!=b&&b!=c&&c!=a){
      g="This is a Scalene Triangle.";
    }
    else{
      g="This is a Isosceles Triangle.";
    }
    return g;
  }
  
  public void compareTrinagles(Triangle t){
    
    if(this==t){
      System.out.println("These two triangle objects have the same address.");
    }
    else{
      if(this.a==t.a&&this.b==t.b&&this.c==t.c) {
        System.out.println("Addresses are different but the sides of the triangles are equal.");
      } 
      else if((this.a!=t.a||this.b!=t.b||this.c!=t.c)&&this.pre!= t.pre){
        System.out.println("Addresses, length of the sides and perimeter all are different.");
      } 
      else{
        System.out.println("Only the perimeter of both triangles is equal.");
      }
    }
  }
}
